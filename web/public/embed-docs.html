KeyWitness Embed SDK

Add cryptographic verification badges to any website in seconds.

Quick Start

Option 1: HTML attributes (zero code)

<!-- Load the SDK -->
<script src="https://keywitness.io/embed.js"></script>

<!-- Add a badge anywhere -->
<span data-keywitness="abc123" data-keywitness-style="inline"></span>

The SDK auto-discovers all data-keywitness elements on page load.

Option 2: JavaScript API

<script src="https://keywitness.io/embed.js"></script>
<script>
  // Render a badge
  const badge = KeyWitness.badge('#my-element', 'abc123', {
    style: 'card',    // 'inline' | 'card' | 'floating'
    theme: 'dark',    // 'light' | 'dark' | 'auto'
    onVerified(result) {
      console.log('Verification result:', result);
    }
  });

  // Later: update or destroy
  badge.update('newShortId');
  badge.destroy();
</script>

Option 3: Server-side verification

// Verify by attestation ID
const resp = await fetch('https://keywitness.io/api/verify?id=abc123');
const result = await resp.json();
// { valid: true, version: "v3", deviceId: "...", timestamp: "...", ... }

// Verify raw attestation block
const resp = await fetch('https://keywitness.io/api/verify', {
  method: 'POST',
  headers: { 'Content-Type': 'application/json' },
  body: JSON.stringify({
    attestation: '-----BEGIN KEYWITNESS ATTESTATION-----\n...\n-----END KEYWITNESS ATTESTATION-----'
  })
});
const result = await resp.json();

REST API

GET /api/verify?id={shortId}

Verify an attestation by its short ID. Returns structured verification result.

{
  "shortId": "abc123",
  "url": "https://keywitness.io/v/abc123",
  "valid": true,
  "version": "v3",
  "deviceId": "4BF95BEA-...",
  "timestamp": "2026-03-09T04:07:41.132Z",
  "issuerDID": "did:key:z6Mk...",
  "publicKeyFingerprint": "a1:b2:c3:...",
  "appAttestPresent": true,
  "deviceVerified": true,
  "biometricVerified": false,
  "proofs": [
    { "proofType": "keystrokeAttestation", "valid": true },
    { "proofType": "deviceAttestation", "valid": true }
  ]
}

POST /api/verify

Verify a raw attestation PEM block. Full server-side cryptographic verification.

// Request
{ "attestation": "-----BEGIN KEYWITNESS ATTESTATION-----\n...\n-----END KEYWITNESS ATTESTATION-----" }

// Response: same structure as GET (minus shortId/url/deviceVerified/biometricVerified)

GET /api/oembed?url={url}

oEmbed endpoint for rich embeds in Slack, Discord, CMS tools, etc. Returns a rich oEmbed response with an iframe.

oEmbed & OpenGraph

KeyWitness verification URLs (keywitness.io/v/{id}) include:

  • OpenGraph meta tags for rich previews in social media, Slack, Discord
  • oEmbed discovery for platforms that support it
  • Twitter Card meta tags

Share a verification URL and it automatically shows a rich preview with verification status.

Example: Form That Requires a Seal

This example shows a comment form that won't submit unless the text includes a valid KeyWitness attestation link. Once verified, it shows checkmarks for each verification aspect.

<script src="https://keywitness.io/embed.js"></script>

<form id="verified-form">
  <textarea id="comment" placeholder="Type with KeyWitness keyboard, tap Seal..."></textarea>
  <div id="verification-status"></div>
  <button id="submit-btn" type="submit" disabled>Submit (requires seal)</button>
</form>

<script>
const textarea = document.getElementById('comment');
const statusEl = document.getElementById('verification-status');
const submitBtn = document.getElementById('submit-btn');

let verifiedSnapshot = null; // text content when verification passed
let stopWatching = null;     // stop function from watchById

function resetForm() {
  statusEl.innerHTML = '';
  submitBtn.disabled = true;
  submitBtn.style.background = '#374151';
  submitBtn.style.color = '#6b7280';
  submitBtn.style.cursor = 'not-allowed';
  submitBtn.textContent = 'Submit (requires seal)';
  verifiedSnapshot = null;
  if (stopWatching) { stopWatching(); stopWatching = null; }
}

// SHA-256 hash the text the same way the iOS app does
async function sha256Base64URL(text) {
  const data = new TextEncoder().encode(text);
  const hash = await crypto.subtle.digest('SHA-256', data);
  const bytes = new Uint8Array(hash);
  let binary = '';
  for (const b of bytes) binary += String.fromCharCode(b);
  return btoa(binary).replace(/\+/g, '-').replace(/\//g, '_').replace(/=+$/, '');
}

// Extract the cleartext (text without the seal link)
function extractCleartext(text, match) {
  return text.replace(match[0], '').replace(/\s+$/, '');
}

async function handleResult(result, match) {
  if (result.valid) {
    // Verify the text in the form matches the sealed text
    if (result.cleartextHash) {
      const cleartext = extractCleartext(textarea.value, match);
      const hash = await sha256Base64URL(cleartext);
      if (hash !== result.cleartextHash) {
        resetForm();
        statusEl.innerHTML = '<span style="color: #ef4444">'
          + '✗ Text doesn\\'t match seal — the sealed text is different from what\\'s in this form</span>';
        return;
      }
    }

    // Snapshot the text so we can detect edits
    verifiedSnapshot = textarea.value;

    submitBtn.disabled = false;
    submitBtn.style.background = '#22c55e';
    submitBtn.style.color = '#fff';
    submitBtn.style.cursor = 'pointer';
    submitBtn.textContent = 'Submit';

    const checks = [
      { ok: result.valid, label: 'Signature' },
      { ok: result.deviceVerified, label: 'Real device' },
      { ok: result.biometricVerified, label: 'Face ID' },
    ];
    statusEl.innerHTML = checks.map(c =>
      `<span style="margin-right: 12px; color: ${c.ok ? '#22c55e' : '#6b7280'}">`
      + `${c.ok ? '✓' : '&ndash;'} ${c.label}</span>`
    ).join('');
  } else {
    resetForm();
    statusEl.innerHTML = '<span style="color: #ef4444">✗ Invalid seal</span>';
  }
}

textarea.addEventListener('input', () => {
  // If text was edited after verification, invalidate
  if (verifiedSnapshot !== null && textarea.value !== verifiedSnapshot) {
    resetForm();
    statusEl.innerHTML = '<span style="color: #ef4444">'
      + '✗ Text modified after seal — re-seal to submit</span>';
    return;
  }

  // Look for a KeyWitness link in the text
  const match = textarea.value.match(
    /(?:https?:\/\/)?keywitness\.io\/v\/([A-Za-z0-9]+)(?:#[A-Za-z0-9_-]+)?/
  );
  if (!match) { resetForm(); return; }
  if (stopWatching) return; // already watching this seal

  const shortId = match[1];
  statusEl.innerHTML = '<span style="color: #9ca3af">Verifying...</span>';

  // Start watching — polls every 3s, updates badges live as
  // Face ID / device attestation data arrives on the server
  stopWatching = KeyWitness.watchById(shortId, (result) => {
    handleResult(result, match);
  });
});
</script>

The key parts:

  • Watch the textarea for a keywitness.io/v/... link
  • Call KeyWitness.watchById(shortId, callback) to verify and poll for live updates (Face ID, device attestation) for 30 seconds
  • Hash the textarea text (minus the seal link) and compare against result.cleartextHash to verify the text matches what was sealed
  • Snapshot the text on successful verification — if the user edits it afterward, invalidate and require a new seal
  • Check result.valid, result.deviceVerified, result.biometricVerified, and result.proofs
  • Only enable the submit button when the seal is valid and text matches the sealed content

Content Security Policy

If your site uses CSP headers, add:

frame-src https://keywitness.io;
script-src https://keywitness.io;

Events

The badge iframe communicates with the parent page via postMessage:

window.addEventListener('message', (event) => {
  if (event.origin !== 'https://keywitness.io') return;

  if (event.data.type === 'keywitness:verified') {
    console.log(event.data.shortId, event.data.result);
  }
});
